rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function isAdmin() {
      return request.auth != null &&
        request.auth.token.email in ['admin@interliga.com.br', 'miqueias@interliga.com.br'];
    }
    function souFranqueado() {
      return request.auth != null && exists(/databases/$(database)/documents/franqueados/$(request.auth.uid));
    }
    function minhaCidadeFranqueado() {
      return get(/databases/$(database)/documents/franqueados/$(request.auth.uid)).data.cidade;
    }

    match /corridas/{corridaId} {
      allow read, write: if request.auth != null;
      match /mensagens/{mensagemId} {
        allow read, write: if request.auth != null;
      }
    }

    match /motoristas/{motoristaId} {
      allow read: if request.auth != null;
      allow create: if request.auth != null && request.auth.uid == motoristaId;
      allow update: if request.auth != null && (
        request.auth.uid == motoristaId || isAdmin() ||
        (souFranqueado() && resource.data.cidade == minhaCidadeFranqueado())
      );
    }

    match /passageiros/{uid} {
      allow read: if request.auth != null;
      allow create: if request.auth != null && request.auth.uid == uid;
      allow update: if request.auth != null && (
        request.auth.uid == uid || isAdmin() ||
        (souFranqueado() && resource.data.cidade == minhaCidadeFranqueado())
      );
    }

    match /motoristas_disponiveis/{motoristaId} {
      allow read, write: if request.auth != null;
    }

    match /zonas_risco/{zonaId} {
      allow read: if request.auth != null;
      allow create: if isAdmin() || (souFranqueado() && request.resource.data.cidade == minhaCidadeFranqueado());
      allow update, delete: if isAdmin() || (souFranqueado() && resource.data.cidade == minhaCidadeFranqueado());
    }

    match /regras_horario/{regraId} {
      allow read: if request.auth != null;
      allow create: if isAdmin() || souFranqueado();
      allow update, delete: if isAdmin() || (souFranqueado() && resource.data.cidade == minhaCidadeFranqueado());
    }

    match /precos/{cidadeId} {
      allow read: if request.auth != null;
      allow write: if isAdmin() || (souFranqueado() && minhaCidadeFranqueado() == cidadeId);
    }

    match /repasses_pagos/{motoristaId} {
      allow read: if request.auth != null;
      allow write: if isAdmin() || (
        souFranqueado() &&
        get(/databases/$(database)/documents/motoristas/$(motoristaId)).data.cidade == minhaCidadeFranqueado()
      );
    }

    match /config/{configId} {
      allow read: if request.auth != null;
      allow write: if isAdmin();
    }

    match /franqueados/{uid} {
      allow read: if request.auth != null;
      allow write: if isAdmin();
    }

    match /estabelecimentos/{estabId} {
      allow read: if request.auth != null;
      allow create: if isAdmin() || souFranqueado();
      allow update, delete: if isAdmin() || (souFranqueado() && resource.data.cidade == minhaCidadeFranqueado());
    }

    match /avaliacoes/{avaliacaoId} {
      allow read, write: if request.auth != null;
    }

    match /carteira_transacoes/{transacaoId} {
      allow read, create: if request.auth != null;
    }
  }
}
